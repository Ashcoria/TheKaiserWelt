defined_text = {
	name = GetSIAEngineers
	text = {
		trigger = {
			has_socialist_government = yes
		}
		localization_key = SIA_engineers_INT
	}
	text = {
		trigger = {
			is_in_faction_with = JAP
			JAP = { is_subject = no }
		}
		localization_key = SIA_engineers_JAP
	}
	text = {
		trigger = {
			country_exists = GER
			country_exists = AUS
			AUS = { is_subject = no }
			NOT = {
				has_war_with = GER
				has_war_with = AUS
			}
		}
		localization_key = SIA_engineers_GER_AUS
	}
	text = {
		trigger = {
			country_exists = GER
			NOT = { has_war_with = GER }
		}
		localization_key = SIA_engineers_GER
	}
	text = {
		trigger = {
			country_exists = RUS
			RUS = {
				is_subject = no
				NOT = { has_war_with = SOV }
			}
			NOT = { has_war_with = RUS }
		}
		localization_key = SIA_engineers_RUS
	}
	text = {
		trigger = {
			OR = {
				country_exists = USA
				country_exists = PSA
				country_exists = TEX
				country_exists = NEE
			}
			has_global_flag = USA_civil_war_over_flag
			NOT = {
				has_war_with = USA
				has_war_with = PSA
				has_war_with = TEX
				has_war_with = NEE
			}
		}
		localization_key = SIA_engineers_USA
	}
}
