			MAF = {

		}

			}
		}
		localization_key = MAF_lk_take_all_MZB
	}

			}
		}
		localization_key = MAF_lk_take_part_ANG_MZB

			}
		}
		localization_key = MAF_lk_take_all_ANG_MZB

					}

			}
