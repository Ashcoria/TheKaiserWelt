political_actions = {
	TIB_restore_order_tibet = {
		icon = generic_flags

		allowed = { tag = TIB }

		available = {
			ROOT = { has_stability > 0 }
			FROM = { is_owned_and_controlled_by = ROOT }
		}

		visible = {
			FROM = {
				is_claimed_by = ROOT
				NOT = { is_core_of = ROOT }
				any_neighbor_state = {
					is_owned_and_controlled_by = ROOT
					is_core_of = ROOT
				}
			}
		}

		on_map_mode = map_and_decisions_view
		cost = 100
		days_remove = 120
		fire_only_once = no

		state_target = yes
		targets = { 353 601 604 858 }
		target_trigger = {
			FROM = {
				is_claimed_by = ROOT
				NOT = { is_core_of = ROOT }
			}
		}

		cancel_if_not_visible = yes
		cancel_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision cancelled TIB_restore_order_tibet target: [From.GetName]"
			add_political_power = 100
			add_stability = 0.06
		}

		complete_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision TIB_restore_order_tibet target: [From.GetName]"
			FROM = { add_resistance = 40 }
			add_stability = -0.06
		}

		remove_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision remove TIB_restore_order_tibet target: [From.GetName]"
			add_stability = 0.03
			FROM = {
				add_resistance = -40
				if = {
					limit = { is_claimed_by = ROOT }
					remove_claim_by = ROOT
				}
				add_core_of = ROOT
			}
		}

		ai_will_do = {
			factor = 100
		}
	}
