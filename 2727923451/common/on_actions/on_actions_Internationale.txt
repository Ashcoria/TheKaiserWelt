on_actions = {
	#ROOT is winner #FROM is loser
	on_annex = {
		effect = {
			# Reset the INT alias
			if = {
				limit = { FROM = { tag = INT } }
				FRA = { save_global_event_target_as = KR_internationale_leader }
			}
		}
	}

	# ROOT is capitulated country, FROM is winner
	on_capitulation = {
		effect = {
			# France capitulates
			if = {
				limit = {
					tag = FRA
					NOT = { has_global_flag = FRA_capitulated }
				}
				country_event = { id = capt.40 days = 1 }
			}

			# Move the Internationale leader
			if = {
				limit = { tag = INT }
				FRA_transfer_internationale_leader = yes
			}
		}
	}

	#FROM is war target
	on_declare_war = {
		effect = {
			# CoF events
			if = {
				limit = { tag = FRA }
				FROM = {
					if = {
						limit = { tag = BEL }
						country_event = { id = comfrance.166 days = 2 }
					}
					else_if = {
						limit = { tag = BEL }
						country_event = { id = comfrance.167 days = 2 }
					}
				}
			}
		}
	}

	on_new_term_election = {
		effect = {
			if = {
				limit = { tag = FRA }
				country_event = comfrance.144
				country_event = comfrance.155
				country_event = comfrance.156
			}
		}
	}

	#FROM is country getting invited.
	on_offer_join_faction = {
		effect = {
			if = {
				limit = {
					tag = SOV
					is_faction_leader = yes
					INT = {
						is_faction_leader = yes
						NOT = {
							has_war_with = ROOT
							has_war_with = FROM
						}
					}
				}
				INT = {
					give_guarantee = FROM
					diplomatic_relation = {
						country = FROM
						relation = non_aggression_pact
					}
				}
			}
			else_if = {
				limit = {
					tag = INT
					SOV = {
						is_faction_leader = yes
						NOT = {
							has_war_with = ROOT
							has_war_with = FROM
						}
					}
				}
				SOV = {
					give_guarantee = FROM
					diplomatic_relation = {
						country = FROM
						relation = non_aggression_pact
					}
				}
			}
		}
	}
}
