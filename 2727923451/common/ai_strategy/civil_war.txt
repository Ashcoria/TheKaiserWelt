ignore_PSA_rocky_mountain = {
	allowed = {
		OR = {
			tag = USA
			tag = CSA
			tag = TEX
		}
	}
	enable = {
		country_exists = PSA
		NOT = { has_war_with = PSA }
		has_global_flag = PSA_rockies_ceasefire
	}

	abort_when_not_enabled = yes

	ai_strategy = {
		type = ignore_claim
		id = "PSA"
		value = 1
	}
	ai_strategy = {
		type = conquer
		id = "PSA"
		value = 100
	}
}
