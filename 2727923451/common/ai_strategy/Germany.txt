germany_focus_on_western_front_1 = {
	allowed = {
		tag = GER
	}
	enable = {
		country_exists = INT
		INT = { has_capitulated = no }
	}

	abort_when_not_enabled = yes

	ai_strategy = {
		type = front_unit_request
		country_trigger = {
			OR = {
				tag = FRA
				tag = ENG
				tag = HOL
				tag = BEL
				tag = WLL
				tag = FLN
				tag = SWI
				tag = SRI
			}
			is_neighbor_of = GER
			is_in_faction_with = INT
		}
		value = 200
	}
}
