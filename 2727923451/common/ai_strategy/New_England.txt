USA_friends_with_NEE = {
	allowed = {
		tag = USA
	}
	enable = {
		NOT = { has_war_with = NEE }
		NEE = {
			has_completed_focus = NEE_Expand_NEEF
			has_idea = ACW_support_usa
		}
	}

	abort_when_not_enabled = yes

	ai_strategy = {
		type = befriend
		id = "NEE"
		value = 100
	}
}
