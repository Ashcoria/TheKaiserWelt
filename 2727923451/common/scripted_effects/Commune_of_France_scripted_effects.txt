FRA_phalanstere_effect = {
	if = {
		limit = { NOT = { tag = ROOT } }
		custom_effect_tooltip = tooltip_white_line
		custom_effect_tooltip = if_accepted
	}
	add_opinion_modifier = {
		target = INT
		modifier = KR_approves_of_govt
	}
	if = {
		limit = {
			ENG = {
				NOT = { tag = ROOT }
				has_country_flag = FRA_phanlanstere_member
			}
		}
		add_opinion_modifier = {
			target = ENG
			modifier = KR_approves_of_govt
		}
	}
	if = {
		limit = {
			tag = CEN
			OR = {
				312 = { #Honduras
					is_owned_and_controlled_by = PREV
					NOT = { has_state_flag = FRA_phalanstere_developed }
				}
				313 = { #Guatemala
					is_owned_and_controlled_by = PREV
					NOT = { has_state_flag = FRA_phalanstere_developed }
				}
				314 = { #El Salvador
					is_owned_and_controlled_by = PREV
					NOT = { has_state_flag = FRA_phalanstere_developed }
				}
				316 = { #Costa Rica
					is_owned_and_controlled_by = PREV
					NOT = { has_state_flag = FRA_phalanstere_developed }
				}
				317 = { #Managua
					is_owned_and_controlled_by = PREV
					NOT = { has_state_flag = FRA_phalanstere_developed }
				}
			}
		}
		every_owned_state = {
			limit = {
				OR = {
					state = 312 #Honduras
					state = 313 #Guatemala
					state = 314 #El Salvador
					state = 316 #Costa Rica
					state = 317 #Managua
				}
				NOT = { has_state_flag = FRA_phalanstere_developed }
			}
			FRA_phalanstere_development_in_targeted_state = yes
		}
	}
	else_if = {
		limit = { has_variable = FRA_num_target_states }
		FRA_phalanstere_development_in_random_state = yes
		FRA_phalanstere_infrastructure_effect = yes
		if = {
			limit = { check_variable = { FRA_num_target_states > 1 } }
			FRA_phalanstere_development_in_random_state = yes
			FRA_phalanstere_infrastructure_effect = yes
		}
		if = {
			limit = { check_variable = { FRA_num_target_states > 2 } }
			FRA_phalanstere_development_in_random_state = yes
			FRA_phalanstere_infrastructure_effect = yes
		}
		if = {
			limit = { check_variable = { FRA_num_target_states > 3 } }
			FRA_phalanstere_development_in_random_state = yes
			FRA_phalanstere_infrastructure_effect = yes
		}
	}
	else = {
		FRA_phalanstere_development_in_capital = yes
		FRA_phalanstere_infrastructure_effect = yes
	}
	FRA_add_to_phalanstere = yes
	if = {
		limit = { NOT = { tag = ENG } }
		add_tech_bonus = {
			name = FRA_Develop_Country
			bonus = 1
			uses = 1
			category = industry
		}
	}
}

FRA_phalanstere_development_in_random_state = {
	random_owned_controlled_state = {
		limit = {
			is_core_of = ROOT
			is_in_home_area = yes
			free_building_slots = {
				building = industrial_complex
				size > 0
				include_locked = yes
			}
			is_valid_state_for_development = yes
			NOT = { has_state_flag = FRA_phalanstere_developed }
		}
		set_state_flag = FRA_phalanstere_developed
		add_extra_state_shared_building_slots = 1
		add_building_construction = {
			type = industrial_complex
			level = 1
			instant_build = yes
		}
	}
}

FRA_phalanstere_development_in_targeted_state = {
	if = {
		limit = {
			free_building_slots = {
				building = industrial_complex
				size > 0
				include_locked = yes
			}
			NOT = { has_state_flag = FRA_phalanstere_developed }
		}
		set_state_flag = FRA_phalanstere_developed
		add_extra_state_shared_building_slots = 1
		add_building_construction = {
			type = infrastructure
			level = 1
			instant_build = yes
		}
		add_building_construction = {
			type = industrial_complex
			level = 1
			instant_build = yes
		}
	}
	else = {
		FRA_phalanstere_development_in_random_state = yes
		FRA_phalanstere_infrastructure_effect = yes
	}
}

FRA_phalanstere_infrastructure_effect = {
	random_owned_controlled_state = {
		limit = {
			is_in_home_area = yes
			OR = {
				oil > 0
				rubber > 0
				steel > 0
				aluminium > 0
				tungsten > 0
				chromium > 0
			}
			free_building_slots = { building = infrastructure size > 0 include_locked = yes }
			is_valid_state_for_development = yes
		}
		add_building_construction = {
			type = infrastructure
			level = 1
			instant_build = yes
		}
	}
}

FRA_transfer_internationale_leader = {
	if = {
		limit = {
			OR = {
				any_allied_country = {
					is_actual_major = yes
					has_capitulated = no
					is_subject = no
				}
				SOV = { is_faction_leader = yes }
			}
		}
		get_highest_scored_country_temp = {
			scorer = next_internationale_leader_scorer
			var = next_internationale_leader
		}
		if = {
			limit = {
				NOT = { has_variable = next_internationale_leader }
				SOV = { is_faction_leader = yes }
			}
			set_temp_variable = { next_internationale_leader = SOV }
		}
		if = {
			limit = {
				check_variable = { next_internationale_leader = SOV }
				SOV = { is_faction_leader = yes }
			}
			every_other_country = {
				limit = {
					is_in_faction_with = SOV
					is_subject = no
				}
				set_global_flag = TEMPORARY_DISABLED_FACTION_POPUP
				PREV = {
					add_to_faction = PREV
					hidden_effect = {
						diplomatic_relation = {
							country = PREV
							relation = guarantee
							active = no
						}
						diplomatic_relation = {
							country = PREV
							relation = non_aggression_pact
							active = no
						}
					}
				}
			}
		}
		var:next_internationale_leader = {
			set_faction_leader = yes
			save_global_event_target_as = KR_internationale_leader
			if = {
				limit = {
					OR = {
						NOT = { is_in_faction_with = FRA }
						FRA = { has_capitulated = yes }
					}
					OR = {
						NOT = { is_in_faction_with = ENG }
						ENG = { has_capitulated = yes }
					}
					OR = {
						NOT = { is_in_faction_with = SRI }
						SRI = { has_capitulated = yes }
					}
					NOT = { country_exists = CSA }
				}
				every_other_country = {
					limit = {
						is_ally_with = PREV
						NOT = {
							is_neighbor_of = PREV
							any_neighbor_country = {
								is_ally_with = PREV
								has_capitulated = no
								surrender_progress < 0.8
								is_neighbor_of = PREV.PREV
							}
						}
						OR = {
							has_capitulated = yes
							surrender_progress > 0.8
						}
					}
					leave_faction = yes
				}
			}
		}
	}
	else = {
		### Internationale collapses
		every_other_country = {
			limit = {
				is_in_faction_with = ROOT
				is_subject = no
				has_capitulated = no
			}
			leave_faction = yes
			country_event = capt.4200
		}
		faction_leader = { dismantle_faction = yes }
		FRA = { save_global_event_target_as = KR_internationale_leader }
	}
}

FRA_sabotage_state = {
	if = {
		limit = { non_damaged_building_level = { building = industrial_complex level > 0 } }
		damage_building = {
			type = industrial_complex
			damage = 0.3
		}
	}
	if = {
		limit = { non_damaged_building_level = { building = arms_factory level > 0 } }
		damage_building = {
			type = arms_factory
			damage = 0.3
		}
	}
	if = {
		limit = { non_damaged_building_level = { building = dockyard level > 0 } }
		damage_building = {
			type = dockyard
			damage = 0.4
		}
	}
	if = {
		limit = { non_damaged_building_level = { building = infrastructure level > 0 } }
		damage_building = {
			type = infrastructure
			damage = 0.6
		}
	}
}

FRA_add_to_phalanstere = {
	if = {
		limit = { NOT = { has_country_flag = FRA_phanlanstere_member } }
		set_country_flag = FRA_phanlanstere_member
		add_to_array = { global.phalanstere_members = THIS }
		if = {
			limit = { has_dlc = "Together for Victory" }
			add_to_tech_sharing_group = phalanstere_research
		}
	}
}

FRA_remove_from_phalanstere = {
	if = {
		limit = { has_country_flag = FRA_phanlanstere_member }
		clr_country_flag = FRA_phanlanstere_member
		remove_from_array = { global.phalanstere_members = THIS }
		if = {
			limit = { has_dlc = "Together for Victory" }
			add_to_tech_sharing_group = phalanstere_research
		}
	}
}
