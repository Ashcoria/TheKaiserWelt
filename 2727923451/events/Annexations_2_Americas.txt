country_event = { #Fate of America
	id = annex.2000
	immediate = { log = "[GetDateText]: [Root.GetName]: event annex.2000 FROM = [From.GetName]" }
	title = annexation_USA_CONQ
	desc = annexation_USA_CONQ_desc
	picture = GFX_report_event_generic_conference2

	is_triggered_only = yes

	immediate = {
		hidden_effect = {
			add_to_temp_array = { core_removal_array = USA }
			add_to_temp_array = { core_removal_array = CSA }
			add_to_temp_array = { core_removal_array = PSA }
			add_to_temp_array = { core_removal_array = TEX }
			remove_all_cores_and_claims_of_tags = yes
			if = {
				limit = { is_russian_tag = yes }
				add_state_claim = 463
			}
			every_state = {
				limit = {
					OR = {
						is_mainland_america = yes
						state = 463 #Alaska
					}
				}
				add_core_of = USA
			}
			every_state = {
				limit = { is_american_pacific_islands = yes }
				add_core_of = HAW
			}
			USA = {
				if = {
					limit = { has_country_leader = { name = "Douglas MacArthur" } }
					kill_ideology_leader = paternal_autocrat
					set_party_name = {
						ideology = paternal_autocrat
						name = USA_paternal_autocrat_party
						long_name = USA_paternal_autocrat_party_long
					}
				}
				if = {
					limit = { has_country_leader = { name = "Huey Long" ruling_only = no } }
					kill_ideology_leader = authoritarian_democrat
					kill_ideology_leader = paternal_autocrat
					kill_ideology_leader = national_populist
				}
			}
			CSA = {
				if = {
					limit = { has_country_leader = { name = "Jack Reed" } }
					kill_ideology_leader = totalist
					kill_ideology_leader = syndicalist
					kill_ideology_leader = radical_socialist
				}
			}
		}
	}

	option = { #Liberate USA
		name = annex.2000.a
		trigger = {
			has_socialist_government = no
		}
		ai_chance = {
			base = 60
		}
		set_temp_variable = { tag_to_release = USA }
		release_targeted_tag = yes
	}

	option = { #Liberate CSA
		name = annex.2000.b
		trigger = {
			has_socialist_government = yes
		}
		ai_chance = {
			base = 60
		}
		set_temp_variable = { tag_to_release = CSA }
		every_state = {
			limit = {
				OR = {
					is_mainland_america = yes
					state = 463 #Alaska
				}
			}
			remove_core_of = USA
			add_core_of = CSA
		}
		release_targeted_tag = yes
		var:tag_to_release = {
			unlock_national_focus = CSA_second_convention
			if = {
				limit = { has_government = totalist }
				complete_national_focus = CSA_usas
			}
			else = {
				complete_national_focus = CSA_american_commonwealth
			}
		}
	}

	option = { #Occupy USA
		name = annex.occupation
		ai_chance = { base = 0 }
		occupy_massive_country = yes
		set_country_flag = USA_CONQ
	}
}
