country_event = {#General Justo Passes Away
	id = arg_flavour.12
	immediate = { log = "[GetDateText]: [Root.GetName]: event arg_flavour.12 FROM = [From.GetName]" }
	title = arg_flavour.12.t
	trigger = {
		OR = {
			has_character = ARG_augustin_justo
			has_country_flag = FOP_JUSTO_SPARED
			has_country_flag = FOP_LIB_JUSTO_NOT_SPARED
		}
	}
	desc = {
		text = arg_flavour.12.d1
		trigger = {
			tag = ARG
			ARG = {
				has_country_flag = FOP_LIB_JUSTO_SPARED
			}
		}
	}
	desc = {
		text = arg_flavour.12.d1
		trigger = {
			tag = ARG
			ARG = {
				has_country_flag = FOP_LIB_JUSTO_NOT_SPARED
			}
		}
	}
	desc = {
		text = arg_flavour.12.d3
		trigger = {
			tag = FOP
			FOP = {
				has_country_flag = FOP_JUSTO_SPARED
			}
		}
	}


	fire_only_once = yes

	is_triggered_only = yes

	option = {#The end of a patriot
		name = arg_flavour.12.a
		ai_chance = {
			base = 80
		}
		if = {
			limit = { has_character = ARG_augustin_justo }
			effect_tooltip = { retire_character = ARG_augustin_justo }
			hidden_effect = { ARG_augustin_justo = { retire = yes } }
		}
	}
}
