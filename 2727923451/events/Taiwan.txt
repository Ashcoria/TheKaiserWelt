add_namespace = taiwan

# Ask to join Japan/China
country_event = {
	id = taiwan.1
	immediate = { log = "[GetDateText]: [Root.GetName]: event taiwan.1 FROM = [From.GetName]" }
	title = taiwan.1.t

	desc = {
		text = taiwan.1.d.china
		trigger = {
			has_event_target = china
		}
	}
	desc = {
		text = taiwan.1.d.nochina
		trigger = {
			NOT = { has_event_target = china }
		}
	}

	picture = GFX_report_event_spain_civil_war_soldiers

	trigger = {
		tag = TAI
		has_war = no
		country_exists = JAP
		NOT = { has_war_with = JAP }
		if = {
			limit = { is_subject = yes }
			JAP = {
				is_subject_of = ROOT.overlord
				has_capitulated = no
			}
		}
		else = {
			JAP = {
				is_subject = no
				has_capitulated = no
			}
		}
	}

	fire_only_once = yes

	mean_time_to_happen = {
		days = 3
	}

	immediate = {
		if = {
			limit = { is_subject = yes }
			if = {
				limit = {
					608 = {
						is_capital = yes
						owner = {
							is_han_chinese_tag = yes
							is_subject_of = ROOT.overlord
							has_capitulated = no
						}
					}
				}
				var:608.owner = { save_event_target_as = china }
			}
			else_if = {
				limit = {
					613 = {
						is_capital = yes
						owner = {
							is_han_chinese_tag = yes
							is_subject_of = ROOT.overlord
							has_capitulated = no
						}
					}
				}
				var:613.owner = { save_event_target_as = china }
			}
			else_if = {
				limit = {
					592 = {
						is_capital = yes
						owner = {
							is_han_chinese_tag = yes
							is_subject_of = ROOT.overlord
							has_capitulated = no
						}
					}
				}
				var:592.owner = { save_event_target_as = china }
			}
		}
		else = {
			if = {
				limit = {
					608 = {
						is_capital = yes
						owner = {
							is_han_chinese_tag = yes
							is_valid_peace_target = yes
							has_capitulated = no
						}
					}
				}
				var:608.owner = { save_event_target_as = china }
			}
			else_if = {
				limit = {
					613 = {
						is_capital = yes
						owner = {
							is_han_chinese_tag = yes
							is_valid_peace_target = yes
							has_capitulated = no
						}
					}
				}
				var:613.owner = { save_event_target_as = china }
			}
			else_if = {
				limit = {
					592 = {
						is_capital = yes
						owner = {
							is_han_chinese_tag = yes
							is_valid_peace_target = yes
							has_capitulated = no
						}
					}
				}
				var:592.owner = { save_event_target_as = china }
			}
		}
	}

	option = {
		name = taiwan.1.a
		trigger = { is_subject = no }
		ai_chance = { base = 40 }
		JAP = { country_event = { id = taiwan.2 hours = 8 } }
	}

	option = {
		name = taiwan.1.b
		trigger = { is_subject = yes }
		ai_chance = { base = 40 }
		JAP = { save_event_target_as = taiwan_union_target }
		overlord = { country_event = { id = taiwan.3 hours = 8 } }
	}

	option = {
		name = taiwan.1.c
		trigger = {
			has_event_target = china
			is_subject = no
		}
		ai_chance = { base = 60 }
		event_target:china = { country_event = { id = taiwan.2 hours = 8 } }
	}
	option = {
		name = taiwan.1.d
		trigger = {
			has_event_target = china
			is_subject = yes
		}
		ai_chance = { base = 60 }
		event_target:china = { save_event_target_as = taiwan_union_target }
		overlord = { country_event = { id = taiwan.2 hours = 8 } }
	}

	option = {
		name = taiwan.1.e
		ai_chance = { base = 0 }
		add_political_power = -100
	}
}
