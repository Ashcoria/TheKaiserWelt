country_event = { #Fate of Central Asia
	id = annex.3000
	immediate = { log = "[GetDateText]: [Root.GetName]: event annex.3000 FROM = [From.GetName]" }
	title = annexation_CA_CONQ
	desc = annexation_CA_CONQ_desc
	picture = GFX_report_event_SIK_scenery

	immediate = {
		hidden_effect = {
			add_to_temp_array = { core_removal_array = TRK }
			remove_all_cores_and_claims_of_tags = yes
			every_state = {
				limit = { is_turkestan = yes }
				add_core_of = TRK
			}
			TRK = { drop_cosmetic_tag = yes }
		}
	}

	is_triggered_only = yes

	option = { #Liberate an united Central Asia
		name = annex.3000.a
		ai_chance = {
			base = 20
			modifier = {
				factor = 0
				is_russian_tag = yes
			}
			modifier = {
				factor = 0
				var:571.owner = { #Omsk
					is_russian_tag = yes
					is_ally_with = ROOT
				}
			}
		}
		every_state = {
			limit = {
				is_central_asia = yes
				NOT = {
					is_claimed_by = var:tag_to_release
					is_core_of = var:tag_to_release
				}
			}
			add_claim_by = TRK
		}
		set_temp_variable = { tag_to_release = TRK }
		release_targeted_tag_with_claims = yes
		var:tag_to_release = { set_cosmetic_tag = CA_united }
	}

	option = { #give to Russia
		name = annex.3000.b
		trigger = {
			is_russian_tag = no
			var:571.owner = { #Omsk
				is_russian_tag = yes
				is_ally_with = ROOT
			}
		}
		ai_chance = {
			base = 80
			modifier = {
				factor = 0
				is_german_tag = yes
			}
		}
		every_owned_state = {
			limit = {
				region = 136 #Central Asia
				is_claim_or_core_of_ROOT = no
			}
			var:571.owner = { transfer_state = PREV }
		}
		recheck_annexations = yes
	}

	option = { #divide
		name = annex.3000.c
		ai_chance = {
			base = 80
			modifier = {
				factor = 0
				is_russian_tag = yes
			}
			modifier = {
				factor = 0
				var:571.owner = { #Omsk
					is_russian_tag = yes
					is_ally_with = ROOT
				}
			}
		}
		set_country_flag = { flag = CA_CONQ days = 100 value = 1 }
		recheck_annexations_immediate = yes
	}

	option = { #Integrate
		name = annex.integration
		trigger = {
			is_russian_tag = yes
		}
		ai_chance = {
			base = 100
		}
		every_state = {
			limit = {
				is_central_asia = yes
				is_claim_or_core_of_ROOT = no
				NOT = {
					is_claim_or_core_of_owner = yes
					owner = {
						NOT = { tag = ROOT }
						is_ally_with = ROOT
					}
				}
			}
			add_claim_by = ROOT
		}
		add_political_power = -120
		add_to_array = { RUS_integration_array = ALO }
		add_to_array = { RUS_integration_array = TRK }
		add_to_array = { RUS_integration_array = BUK }
		add_to_array = { RUS_integration_array = KHI }
		recheck_annexations = yes
	}

	option = { #occupy
		name = annex.occupation
		trigger = {
			is_russian_tag = no
		}
		ai_chance = { base = 0 }
		set_country_flag = CA_CONQ
		set_country_flag = ALO_CONQ
		set_country_flag = TRK_CONQ
		set_country_flag = BUK_CONQ
		set_country_flag = KHI_CONQ
		occupy_massive_country = yes
	}
}
