#Paraguay
country_event = { #Paraguay debates intervention
	id = southern.4
	immediate = { log = "[GetDateText]: [Root.GetName]: event southern.4 FROM = [From.GetName]" }
	title = southern.4.t
	desc = southern.4.d

	is_triggered_only = yes

	trigger = {
		ARG = {
			has_war_with = CHL
			NOT = { is_in_faction_with = PAR }
		}
	}

	option = {
		ai_chance = {
			base = 100
		}
		name = southern.4.a
		ARG = {
			country_event = southern.5
		}
		declare_war_on = {
			target = ARG
			type = annex_everything
		}
		add_state_claim = 1074
		hidden_effect = {
			if = {
				limit = {
					BOL = {
						exists = yes
					}
				}
				BOL = {
					country_event = {
						id = southern.6
						days = 2
					}
				}
			}
			else_if = {
				limit = {
					PRU = {
						exists = yes
						has_cosmetic_tag = PBC
					}
				}
				PRU = {
					country_event = {
						id = southern.6
						days = 2
					}
				}
			}
		}
	}
	option = {
		ai_chance = {
			base = 0
		}
		name = southern.4.b
	}
}
