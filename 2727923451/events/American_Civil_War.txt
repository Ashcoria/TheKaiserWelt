country_event = {
	id = amcivwar.2400 #system event to end the civil war
	immediate = { log = "[GetDateText]: [Root.GetName]: event amcivwar.2400 FROM = [From.GetName]" }
	hidden = yes

	fire_only_once = yes
	is_triggered_only = yes

	trigger = {
		exists = yes
		has_global_flag = USA_civil_war_begun_flag
		NOT = {
			has_war_with = CAN
			has_war_with = NEE
			has_war_with = MEX
			has_country_flag = ACW_has_deal_with_devil
		}
		if = {
			limit = { NOT = { tag = CSA } }
			NOT = { country_exists = CSA }
		}
		if = {
			limit = { NOT = { tag = USA } }
			NOT = { country_exists = USA }
		}
		if = {
			limit = { NOT = { tag = TEX } }
			NOT = { country_exists = TEX }
		}
		if = {
			limit = { NOT = { tag = PSA } }
			OR = {
				NOT = { country_exists = PSA }
				is_in_faction_with = PSA
			}
		}
	}

	option = {
		USA_civil_war_ends = yes
		USA_demobilize_forces = yes
		remove_ACW_claims = yes

		### Events for other tags
		BRA = { country_event = { id = bra.110 days = 15 random_days = 20 } }
		LEC = { country_event = { id = lec.150 days = 5 random_days = 10 } }
		HAI = { country_event = { id = hai.21 days = 5 random_days = 5 } }

		if = {
			limit = {
				OR = {
					country_exists = USA
					country_exists = CSA
					country_exists = TEX
					country_exists = PSA
				}
			}
			every_country = {
				limit = {
					OR = {
						tag = USA
						tag = CSA
						tag = PSA
						tag = TEX
						AND = {
							tag = NEE
							has_country_flag = NEE_entered_the_civil_war
						}
					}
				}
				country_event = amcivwar.24
			}
		}
		else_if = {
			limit = { country_exists = NEE }
			NEE = { country_event = amcivwar.24 }
		}
		else = {
			var:361.owner = { country_event = amcivwar.24 }
		}
	}
}
