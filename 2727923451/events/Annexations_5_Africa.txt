	id = annex.5011
	immediate = { log = "[GetDateText]: [Root.GetName]: event annex.5011 FROM = [From.GetName]" }
	title = annexation_LIB_CONQ
	desc = annexation_LIB_CONQ_desc
	picture = GFX_report_event_generic_conference2

	is_triggered_only = yes

	immediate = {
		hidden_effect = {
			add_to_temp_array = { core_removal_array = LIB }
			remove_all_cores_and_claims_of_tags = yes
			LIB = {
				add_state_core = 298
				clr_country_flag = SIE_CONQ
				clr_country_flag = IVO_CONQ
				clr_country_flag = GNA_CONQ
				clr_country_flag = GHA_CONQ
				clr_country_flag = TOG_CONQ
			}
		}
	}

	option = { #liberate
		name = annex.5011.a
		ai_chance = {
			base = 100
		}

		trigger = {
			OR = {
				NOT = { has_government = authoritarian_democrat }
				tag = GER
				tag = MAF
			}
		}

		hidden_effect = {
			if = {
				limit = {
					"LIB" = {
						has_cosmetic_tag = LIB_AUTHDEM
					}
				}
				"LIB" = {
					set_party_name = {
						ideology = authoritarian_democrat
						long_name = "Deutscher Liberian Bund"
						name = "DLB"
					}
				}
			}
		}

		set_temp_variable = { tag_to_release = LIB }
		release_targeted_tag = yes
	}

	option = { #liberate authdem (anybody but Germany and Mittelafrika)
		name = annex.5011.b
		ai_chance = {
			base = 100
		}

		trigger = {
			NOT = {
				tag = GER
				tag = MAF
			}
			has_government = authoritarian_democrat
		}

		set_temp_variable = { tag_to_release = LIB }
		release_targeted_tag = yes
		hidden_effect = {
			if = {
				limit = { has_government = authoritarian_democrat }
				"LIB" = {
					set_cosmetic_tag = LIB_AUTHDEM
					set_party_name = {
						ideology = authoritarian_democrat
						long_name = "Authoritarian Dem."
						name = "Authoritarian Dem."
					}
				}
			}
		}
	}

	option = { #occupy
		name = annex.occupation
		ai_chance = {
			base = 0
		}
		set_country_flag = LIB_CONQ
		occupy_small_country = yes
	}
