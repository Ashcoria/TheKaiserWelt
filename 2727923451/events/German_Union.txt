	desc = gru.1.desc
	picture = GFX_report_event_generic_big_city

		OR = {
			tag = GRU

			tag = NGF
		}

		}

	}

			base = 40
		}
		set_province_name = { id = 11467 name = "Eisenhüttenstadt" }

	}
