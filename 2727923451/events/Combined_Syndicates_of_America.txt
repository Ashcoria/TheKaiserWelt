country_event = {
	id = csa.35 #hidden event to fire the correct election (donotlog)
	hidden = yes

	trigger = {
		NOT = { has_government = totalist }
		check_variable = { CSA_election = 3 }
	}

	fire_only_once = yes
	is_triggered_only = yes

	option = {
		if = {
			limit = { has_country_leader = { character = CSA_elizabeth_gurley_flynn } }
			country_event = csa.36 #Zeidler vs Amter
		}
		else_if = {
			limit = { has_country_leader = { character = CSA_daniel_hoan } }
			country_event = csa.37 #Allison vs Hoopes
		}
		else_if = {
			limit = { has_country_leader = { character = CSA_eleanor_roosevelt } }
			country_event = csa.38 #Roosevelt vs Amter
		}
		else_if = {
			limit = { has_country_leader = { character = CSA_frank_johns } }
			country_event = csa.39 #Johns vs Amter
		}
		else_if = {
			limit = { has_country_leader = { character = CSA_upton_sinclair } }
			country_event = csa.40 #Cannon vs Johnson
		}
		else_if = {
			limit = { has_country_leader = { character = CSA_elmer_allison } }
			country_event = csa.41 #Allison vs Johnson
		}
		else_if = {
			limit = { has_country_leader = { character = CSA_john_llewellyn_lewis } }
			country_event = csa.42 #Cannon vs Lewis
		}
		add_to_variable = { CSA_election = 1 }
	}
}
